[
  CREATE VIEW book_with_genres AS
  SELECT 
    book.id, 
    book.book_name, 
    GROUP_CONCAT(genre.genre_name) AS genres
  FROM book
  JOIN books_genres ON book.id = books_genres.book_id
  JOIN genre ON books_genres.genre_id = genre.id
  GROUP BY book.id;
]
